```bash
# 最新のPrometheusをダウンロード
$ open https://prometheus.io/download/

# Prometheusを起動
$ ~/Downloads/prometheus-2.33.1.darwin-amd64/prometheus --config.file=prometheus.yml

# ビルトインのブラウザを開く
$ open http://localhost:9090/
```
